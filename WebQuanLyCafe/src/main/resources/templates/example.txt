<script>
    document.addEventListener('DOMContentLoaded', function() {
        const mainOrderBtn = document.getElementById('mainOrderBtn');
        const btnText = mainOrderBtn.querySelector('span');
        const progressLine = document.getElementById('progressLine');
        
        const milestones = [
            { el: document.getElementById('milestone-1'), label: document.getElementById('milestone-1').parentElement.querySelector('.milestone-label'), width: '10%' },
            { el: document.getElementById('milestone-2'), label: document.getElementById('milestone-2').parentElement.querySelector('.milestone-label'), width: '40%' },
            { el: document.getElementById('milestone-3'), label: document.getElementById('milestone-3').parentElement.querySelector('.milestone-label'), width: '70%' },
            { el: document.getElementById('milestone-4'), label: document.getElementById('milestone-4').parentElement.querySelector('.milestone-label'), width: '100%' }
        ];

        let currentState = 0; // 0: Not started, 1-4: Steps

        mainOrderBtn.addEventListener('click', function() {
            if (currentState === 4) {
                resetUI();
            } else {
                currentState++;
                updateProgress(currentState);
            }
        });

        function updateProgress(state) {
            const currentMilestone = milestones[state - 1];
            
            // Move progress line
            progressLine.style.width = currentMilestone.width;

            // Activate milestone visual with slight delay for the line to reach
            setTimeout(() => {
                currentMilestone.el.classList.add('milestone-active');
                
                // Special animation for milestones
                if (state === 4) {
                    currentMilestone.el.classList.add('animate-success-pulse');
                    // Update label with check icon for final state if desired, but request asked for 4th milestone text
                } else {
                    currentMilestone.el.classList.add('animate-bounce-once');
                }
                
                currentMilestone.label.classList.remove('opacity-40');
                currentMilestone.label.classList.add('text-secondary', 'opacity-100');
                
                // Update Button UI based on state
                if (state === 1) {
                    mainOrderBtn.classList.add('btn-ordered');
                    btnText.textContent = 'HỦY ĐƠN';
                } else if (state === 2) {
                    btnText.textContent = 'ĐANG THỰC HIỆN...';
                    mainOrderBtn.classList.remove('btn-ordered');
                    mainOrderBtn.classList.add('bg-primary');
                } else if (state === 3) {
                    btnText.textContent = 'HOÀN TẤT - CHỜ THANH TOÁN';
                } else if (state === 4) {
                    btnText.textContent = 'ĐÃ THANH TOÁN - CẢM ƠN!';
                    mainOrderBtn.classList.remove('bg-primary');
                    mainOrderBtn.classList.add('bg-secondary');
                }
            }, 800);
        }

        function resetUI() {
            currentState = 0;
            mainOrderBtn.classList.remove('btn-ordered', 'bg-secondary', 'bg-primary');
            mainOrderBtn.classList.add('bg-primary');
            btnText.textContent = 'Đặt đơn';
            progressLine.style.width = '0%';
            
            milestones.forEach(m => {
                m.el.classList.remove('milestone-active', 'animate-bounce-once', 'animate-success-pulse');
                m.label.classList.add('opacity-40');
                m.label.classList.remove('text-secondary', 'opacity-100');
            });
        }
    });
</script>
